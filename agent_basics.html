"""
Minimal AI Agent — demonstrates 3 of 4 core concepts:
  1. LLM Calling      — client.messages.create()
  2. Tool Calling     — define tools, handle tool_use blocks
  3. Loop Iteration   — agentic loop: keep calling until end_turn
"""

import json
import anthropic

# ── 1. LLM CALLING ────────────────────────────────────────────────────────────
client = anthropic.Anthropic()  # reads ANTHROPIC_API_KEY from env

# ── 2. TOOL DEFINITION ────────────────────────────────────────────────────────
def get_weather(location: str) -> str:
    """Fake weather tool — replace with a real API call."""
    return f"72°F and sunny in {location}"

TOOLS = [
    {
        "name": "get_weather",
        "description": "Get the current weather for a city.",
        "input_schema": {
            "type": "object",
            "properties": {
                "location": {"type": "string", "description": "City name"}
            },
            "required": ["location"],
        },
    }
]

TOOL_REGISTRY = {"get_weather": get_weather}





# ── 3. LOOP ITERATION (agentic loop) ────────────────────────────────────────────
def run_agent(user_prompt: str) -> str:
    messages = [{"role": "user", "content": user_prompt}]

    while True:
        # --- LLM call ---
        response = client.messages.create(
            model="claude-opus-4-6",
            max_tokens=1024,
            tools=TOOLS,
            messages=messages,
        )

        # Loop exit: Claude is done
        if response.stop_reason == "end_turn":
            return next(b.text for b in response.content if b.type == "text")

        # --- Tool call handling ---
        # Append Claude's response (may contain tool_use blocks)
        messages.append({"role": "assistant", "content": response.content})

        # Execute every tool Claude requested, collect results
        tool_results = []
        for block in response.content:
            if block.type == "tool_use":
                fn = TOOL_REGISTRY[block.name]
                result = fn(**block.input)          # call the local function
                print(f"  [tool] {block.name}({block.input}) → {result}")
                tool_results.append({
                    "type": "tool_result",
                    "tool_use_id": block.id,        # must match the request id
                    "content": result,
                })

        # Feed results back so the loop continues
        messages.append({"role": "user", "content": tool_results})


if __name__ == "__main__":
    answer = run_agent("What's the weather like in Paris and Tokyo?")
    print("\nFinal answer:", answer)

# ── Uncertainty → Certainty ───────────────────────────────────────────────────
# The tool definition declares exact input parameters and their types.
# Claude reads that schema and must reply with a concrete value for every
# required field — no ambiguity allowed. The Agent then strictly validates
# each value: range checks, syntax & semantic checks, and optionally a
# second LLM call to verify correctness before the tool is executed.
#
# 这是 LLM 从「不确定」走向「确定」的关键时刻。工具定义声明了每个输入参数及其
# 类型；Claude 读取该 schema 后，必须为每个必填字段给出具体值，不允许模糊回答。
# Agent 随后对每个参数严格验证：值域检查、语法与语义检查，以及可选的第二次
# LLM 调用来确认参数正确性，再执行工具。
